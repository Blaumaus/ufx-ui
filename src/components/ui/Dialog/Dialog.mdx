## Basic usage

<Playground>
  {() => {
    const [isOpen, setIsOpen] = React.useState(false)
    return (
      <>
        <button
          type='button'
          onClick={() => setIsOpen(!isOpen)}
        >
          Toggle
        </button>
        <Dialog
          isOpen={isOpen}
          onClose={() => setIsOpen(false)}
          title='My Dialog'
        >
          My Dialog Content
        </Dialog>
      </>
    )
  }}
</Playground>
